// Mappa lettere a numeri (sistema pitagorico)
function letteraToNumero(lettera) {
const mappa = {
'A': 1, 'B': 2, 'C': 3, 'D': 4, 'E': 5,
'F': 6, 'G': 7, 'H': 8, 'I': 9,
'J': 1, 'K': 2, 'L': 3, 'M': 4, 'N': 5,
'O': 6, 'P': 7, 'Q': 8, 'R': 9,
'S': 1, 'T': 2, 'U': 3, 'V': 4, 'W': 5,
'X': 6, 'Y': 7, 'Z': 8
};
return mappa[lettera] || 0;
}
// Riduce a cifra singola (tranne numeri maestri)
function riduciANumeroMaestro(numero) {
if ([11, 22, 33].includes(numero)) return numero;
while (numero > 9) {
numero = Math.floor(numero / 10) + (numero % 10);
}
return numero;
}
// Calcola il Percorso di Vita (data di nascita)
function calcolaPercorsoVita(data) {
const numeri = data.replace(/\D/g, '');
let somma = 0;
for (let num of numeri) {
somma += parseInt(num);
}
return riduciANumeroMaestro(somma);
}
// Calcola il Numero del Destino (nome + cognome)
function calcolaDestino(nomeCompleto) {
let somma = 0;
for (let lettera of nomeCompleto.toUpperCase().replace(/[^A-Z]/g, '')) {
somma += letteraToNumero(lettera);
}
return riduciANumeroMaestro(somma);
}
// Trova numeri karmici (da ebook pag. 29)
function trovaNumeriKarmici(nomeCompleto, dataNascita) {
const karmici = [13, 14, 16, 19];
const presenti = [];
const str = nomeCompleto + dataNascita;
// Cerca sequenze numeriche karmiche
for (let num of karmici) {
if (str.includes(num.toString())) {
presenti.push(num);
}
}
return presenti.length ? presenti : ["Nessun numero karmico rilevante"];
}
// Trova numeri mancanti (1-9)
function trovaNumeriMancanti(nomeCompleto, dataNascita) {
const presenti = new Set();
const str = (nomeCompleto + dataNascita).replace(/\D/g, '');
for (let char of str) {
if (/[1-9]/.test(char)) {
presenti.add(parseInt(char));
}
}
const mancanti = [];
for (let i = 1; i <= 9; i++) {
if (!presenti.has(i)) {
mancanti.push(i);
}
}
return mancanti.length ? mancanti : ["Completo (tutti i numeri da 1 a 9 sono
presenti)"];
}
